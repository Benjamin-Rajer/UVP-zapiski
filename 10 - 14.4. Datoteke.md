Zdaj pa dajmo nekaj funkcij napisati z datotekami:

```python
def prestej_besede(datoteka):
	slovar = {}
	with open(datoteka) as dat: # alternativen način odpiranja datotek
		# pišemo z zamikom
		vsebina = dat.read()
		# moramo še očistiti besede da odstranimo vejice, pike itd.
		vsebina = vsebina.replace(",","")
		vsebina = vsebina.replace(".","")
		vsebina = vsebina.replace(";","")
		vsebina = vsebina.replace("\n"," ")
		vsebina = vsebina.replace("?","")
		
		besede = vsebina.split() # razdelimo besede
		for beseda in besede:
			if beseda = "":
				continue
			beseda = beseda.lower() # 
			if beseda not in slovar:
				slovar[beseda] = 1
			else:
				slovar[beseda] += 1
				
	return slovar
	
	# potem lahko še sortiramo slovar
		 
```

Še eno podobno napišimo: zanimajo nas datoteke po velikosti.

```python
import os

def poisci_datoteke(lokacija):
	datoteke = []
	vsebina = os.listdir(lokacija)
	vsebina.sort()
	for j in vsebina:
		if j[0] == "." or j[:2] == "__":
			continue 
			
		je_datoteka = os.path.isfile(os.path.join(lokacija, j)) 
		
		if je_datoteka:
			velikost = os.path.getsize(os.path.join(lokacija, j))
			datoteke.append((j, velikost))
		else:
			datoteke_v_dir = poisci_datoteke(os.path.join(lokacija, j)) # če j ni datoteka, je mapa in gremo vanjo
		
		datoteke.extend(datoteke_v_dir)
	
	datoteke.sort(key=lambda x: -x[1]) # to bo dalo največje na začetek
	
	return datoteke
```

def uredi(ime_slike):
	slika = Image.open(ime_slike)
	nova = Image.new("RGB", (slika.width, slika.height))
	
	for i in range(slika.width):
		for j in range(slika.height):
			nova.putpixel((i, j), slika.getpixel(((slika.width - 1 -i, j)))
			
	nova.save("nova" + ime_slike)
